 16 /**
 17  * Abstraction for the Spire API
 18  *
 19  * @class Collection of API methods
 20  *
 21  * @example
 22  * var api = new API(options);
 23  *
 24  * @constructor
 25  * @param {object} spire Spire object
 26  * @param {object} [opts] Options
 27  * @param {string} [opts.url] Spire api url
 28  * @param {string} [opts.version] Version of the Spire api to use
 29  * @param {string} [opts.timeout] Timeout for requests
 30  */
 31 function API(spire, opts) {
 32   this.spire = spire;
 33 
 34   opts = opts || {};
 35   this.url = opts.url || 'https://api.spire.io';
 36   this.version = opts.version || '1.0';
 37   this.timeout = opts.timeout || 30 * 1000;
 38 
 39   this.description = null;
 40   this.schema = null;
 41 }
 42 
 43 module.exports = API;

 53 /**
 54  * Discovers urls from Spire API.  Since this description does not change often,
 55  * we only make the request once and cache the result for subsequent calls.
 56  *
 57  * @example
 58  * api.discover(function (err, discovered) {
 59  *   if (!err) {
 60  *     // ...
 61  *   }
 62  * });
 63  *
 64  * @param {function(err, discovered)} cb Callback
 65  */
 66 API.prototype.discover = function (cb) {
 67   var api = this;
 68 
 69   if (this.description) {
 70     return process.nextTick(function () {
 71       cb(null, api.description);
 72     });
 73   }
 74 
 75   this.request('discover', function (err, description) {
 76     if (err) return cb(err);
 77     api.description = description;
 78     api.schema = description.schema[api.version];
 79     cb(null, description);
 80   });
 81 };

 83 /**
 84  * Creates a spire session from an account secret.
 85  *
 86  * @param {string} secret The acccount secret
 87  * @param {function(err)} cb Callback
 88  */
 89 API.prototype.createSession = function (secret, cb) {
 90   var api = this;
 91   this.discover(function (err) {
 92     if (err) return cb(err);
 93     api.request('create_session', secret, function (err, sessionData) {
 94       if (err) return cb(err);
 95       var session = new Session(api.spire, sessionData);
 96       cb(null, session);
 97     });
 98   });
 99 };

172 /**
173  * Get Account from url and capabilities.
174  *
175  * Use this method to get the account without starting a spire session.
176  *
177  * If you have a spire session, you should use <code>spire.session.account</code>.
178  *
179  * @example
180  * var spire = new Spire();
181  * spire.api.accountFromUrlAndCapabilities({
182  *   url: account_url,
183  *   capabilities: account_capabilities
184  * }, function (err, account) {
185  *   if (!err) {
186  *     // ...
187  *   }
188  * })
189  *
190  * @param {object} creds Url and Capabilities
191  * @param {string} creds.url Url
192  * @param {string} creds.capabilities Capabilities
193  * @param {function (err, account)} cb Callback
194  */
195 API.prototype.accountFromUrlAndCapabilities = function (creds, cb) {
196   var api = this;
197   this.discover(function (err) {
198     if (err) return cb(err);
199     var account = new Account(api.spire, creds);
200     account.get(cb);
201   });
202 };

308 /**
309  * Returns the MIME type for resourceName.
310  *
311  * @param {string} [name] Name of the resource MIME type to return
312  * @returns {string} MIME type of the resource
313  */
314 API.prototype.mediaType = function (resourceName) {
315   if (!this.schema) {
316     throw "No description object.  Run `spire.api.discover` first.";
317   }
318 
319   if (!this.schema[resourceName]) {
320     throw "No schema for resource " + resourceName;
321   }
322 
323   return this.schema[resourceName].mediaType;
324 };

326 /**
327  * Returns the Authorization header for the resource and method.
328  *
329  * @param {Resource} resource Resource
330  * @param {string} method Method
331  * @returns {string} Authorization header for the resource
332  */
333 API.prototype.authorization = function (method, resource) {
334   return ['Capability', resource.capabilities[method]].join(' ');
335 };
