1 /**
  2  * @fileOverview Member Resource class definition
  3  */
  4 var Resource = require('./resource');
  5 
  6 /**
  7  * Represents a member in the spire api.
  8  *
  9  * @class Member Resource
 10  *
 11  * @constructor
 12  * @extends Resource
 13  * @param {object} spire Spire object
 14  * @param {object} data  Member data from the spire api
 15  */
 16 function Member(spire, data) {
 17   this.spire = spire;
 18   this.data = data;
 19   this.resourceName = 'member';
 20 }
 21 
 22 Member.prototype = new Resource();
 23 
 24 module.exports = Member;
 25 
 26 /**
 27  * Returns the member login.
 28  *
 29  * @returns {string} Member login
 30  */
 31 Member.prototype.login = function () {

 33 };
 34 
 35 /**
 36  * Returns the members profile.
 37  *
 38  * @returns {string} Member profile

 40 Member.prototype.profile = function () {
