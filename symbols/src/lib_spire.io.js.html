 24 /**
 25  * Spire API Client
 26  *
 27  * @class <strong>Spire API Client</strong>
 28  *
 29  * @example
 30  * var spire = new Spire();
 31  *
 32  * @example
 33  * var spire = new Spire({
 34  *   secret: my_account_secret
 35  * });
 36  *
 37  * @constructor
 38  * @param {object} [opts] Options for Spire
 39  * @param {string} [secret] The account API secret.  If you do you not set this, you
 40  * must call one of:
 41  *   * `spire.start(secret, callback)`
 42  *   * `spire.login(email, password, callback)` or
 43  *   * `spire.register(user, callback)
 44  *   before you can start creating channels.
 45  * @param {string} [opts.url] Spire url do use (defaults to 'https://api.spire.io')
 46  * @param {string} opts.version Version of Spire api to use (defaults to '1.0')
 47  * @param {number} opts.timeout Timeout for requests (defaults to 30 seconds)
 48  * @param {boolean} opts.logCurl Log all requests as curl commands (defaults to false)
 49  */
 50 function Spire(opts) {
 51   opts = opts || {};
 52   this.api = new API(this, opts);
 53   this.session = null;
 54   this._opts_secret = opts.secret;
 55   this.shred = new Shred({
 56     logCurl: opts.logCurl
 57   });
 58 }

 62 /**
 63  * Get the account secret.
 64  *
 65  * @returns {string} Account secret
 66  */
 67 Spire.prototype.secret = function () {
 68   this._ensureSession();
 69   if (this.session && this.session.resources && this.session.resources.account) {
 70     return this.session.resources.account.secret();
 71   }
 72   return null;
 73 };

 75 /**
 76  * Start the Spire session with the given account secret.
 77  *
 78  * @example
 79  * var spire = new Spire();
 80  * spire.start(your_api_secret, function (err, session) {
 81  *   if (!err) {
 82  *     // You now have a spire session.
 83  *     // Start creating channels and subscripions.
 84  *   }
 85  * });
 86  *
 87  * @param {string} secret The acccount secret
 88  * @param {function(err)} cb Callback
 89  */
 90 Spire.prototype.start = function (secret, cb) {
 91   var spire = this;
 92   this.api.createSession(secret, function (err, session) {
 93     if (err) return cb(err);
 94     spire.session = session;
 95     cb(null);
 96   });
 97 };

248   });

299 /**
300  * Gets a subscription to the given channels.  Creates the channels and the
301  * subscription if necessary.
302  *
303  * @example
304  * var spire = new Spire();
305  * spire.start(your_api_secret, function (err, session) {
306  *   spire.subscription('mySubscription', ['foo', 'bar'], function (err, subscription) {
307  *     if (!err) {
308  *       // `subscription` is a subscription named 'mySubscription', listening on channels named 'foo' and 'bar'.
309  *     }
310  *   });
311  * });
312  *
313  * @param {string} Subscription name
314  * @param {array or string} channelOrChannels Either a single channel name, or an array of
315  *   channel names to subscribe to
316  * @param {function (err, subscription)} cb Callback
317  */
318 Spire.prototype.subscription = function (name, channelOrChannels, cb) {
319   var channelNames = (typeof channelOrChannels === 'string') ?
320     [channelOrChannels] : channelOrChannels;
321 
322   var spire = this;
323   this._ensureSession(function (err) {
324     if (err) return cb(err);
325     async.forEach(
326       channelNames,
327       function (channelName, innerCB) {
328         spire._findOrCreateChannel(channelName, innerCB);
329       },
330       function (err) {
331         if (err) return cb(err);
332         spire._findOrCreateSubscription(name, channelNames, cb);
333       }
334     );
335   });
336 };

386 /**
387  * Creates an new subscription to a channel or channels, and adds a listener.
388  *
389  * @example
390  * var spire = new Spire();
391  * spire.start(your_api_secret, function (err, session) {
392  *   spire.subscribe('myChannel', options, function (messages) {
393  *     // `messages` is array of messages sent to the channel
394  *   }, function (err) {
395  *   // `err` will be non-null if there was a problem creating the subscription.
396  * });
397  *
398  * By default this will get all events from the beginning of time.
399  * If you only want messages created from this point forward, pass { last: 'now' } in the options:
400  *
401  * @example
402  * var spire = new Spire();
403  * spire.start(your_api_secret, { last: 'now' }, function (err, session) {
404  *   spire.subscribe('myChannel', options, function (messages) {
405  *     // `messages` is array of messages sent to the channel
406  *   }, function (err) {
407  *   // `err` will be non-null if there was a problem creating the subscription.
408  * });
409  *
410  * @param {array or string} channelOrChannels Either a single channel name, or an array of
411  *   channel names to subscribe to
412  * @param {object} [options] Options to pass to the listener
413  * @param {function (messages)} listener Listener that will get called with each batch of messages
414  * @param {function (err, subscription)} [cb] Callback
415  */
416 Spire.prototype.subscribe = function (channelOrChannels, options, listener, cb) {
417   var spire = this;
418   if (typeof options === 'function') {
419     cb = listener;
420     listener = options;
421     options = {}
422   }
423 
424 
425   cb = cb || function () {};
426 
427   var name = 'anon-' + Date.now() + '-' + Math.random();
428 
429   this._ensureSession(function (err) {
430     if (err) return cb(err);
431     spire.subscription(name, channelOrChannels, function (err, subscription) {
432       if (err) return cb(err);
433       subscription.addListener('messages', listener);
434       subscription.startListening(options);
435       process.nextTick(function () {
436         cb(null, subscription);
437       });
438     });
439   });
440 };

580 /**
581  * Number of times to retry creating a channel or subscription before giving up.
582  */
583 Spire.prototype.CREATION_RETRY_LIMIT = 5;

585 /**
586  * Returns the channel with name 'name', creating it if necessary.
587  * You should use `Spire.prototype.channel` method instead of this one.  This
588  * method ignores cached channels.
589  *
590  * @param {string} name Channel name
591  * @param {function (err, channel)} cb Callback
592  */
593 Spire.prototype._findOrCreateChannel = function (name, cb) {
594   var spire = this;
595   var creationCount = 0;
596 
597   function createChannel() {
598     creationCount++;
599     spire.session.createChannel(name, function (err, channel) {
600       if (!err) return cb(null, channel);
601       if (err.status !== 409) return cb(err);
602       if (creationCount >= spire.CREATION_RETRY_LIMIT) {
603         return cb(new Error("Could not create channel: " + name));
604       }
605       getChannel();
606     });
607   }
608 
609   function getChannel() {
610     spire.session.channelByName(name, function (err, channel) {
611       if (err && err.status !== 404) return cb(err);
612       if (channel) return cb(null, channel);
613       createChannel();
614     });
615   }
616 
617   this._ensureSession(function (err) {
618     if (err) return cb(err);
619     getChannel();
620   });
621 };
622 
623 /**
624  * Returns the subvscription with name 'name', creating it if necessary.
625  * You should use `Spire.prototype.subscribe` method instead of this one.  This
626  * method ignores cached subscriptions, and does not create the channels.
627  *
628  * @param {string} name Subscription name
629  * @param {string} channelNames Channel names
630  * @param {function (err, subscription)} cb Callback
631  */
632 Spire.prototype._findOrCreateSubscription = function (name, channelNames, cb) {
633   var spire = this;
634   var creationCount = 0;
635 
636   function createSubscription() {
637     creationCount++;
638     spire.session.createSubscription({
639       name: name,
640       channelNames: channelNames
641     }, function (err, sub) {
642       if (!err) return cb(null, sub);
643       if (err.status !== 409) return cb(err);
644       if (creationCount >= spire.CREATION_RETRY_LIMIT) {
645         return cb(new Error("Could not create subscription: " + name));
646       }
647       getSubscription();
648     });
649   }
650 
651   function getSubscription() {
652     spire.session.subscriptionByName(name, function (err, subscription) {
653       if (err && err.status !== 404) return cb(err);
654       if (subscription) return cb(null, subscription);
655       createSubscription();
656     });
657   }
658 
659   this._ensureSession(function (err) {
660     if (err) return cb(err);
661     createSubscription();
662   });
663 };

665 function NoSessionError(message) {
666   this.name = "No Session Error";
667   this.message = message ||
668     "You need a Spire session to do that.\n" +
669     "Call one of:\n" +
670     "spire.start(callback)\n" +
671     "spire.login(email, password, callback)\n" +
672     "spire.register(user, callback)\n";
673 }

675 NoSessionError.prototype = new Error();
676 NoSessionError.constructor = NoSessionError;
677 
678 Spire.prototype._ensureSession = function (cb) {
679   if (this.session) {
680     if (cb) return cb(null);
681     return;
682   }
683 
684   if (!this._opts_secret) {
685     var noSessionError = new NoSessionError();
686     if (cb) return cb(noSessionError);
687     throw noSessionError;
688   }
689 
690   if (!cb) {
691     throw new NoSessionError();
692   }
693 
694   this.start(this._opts_secret, cb);
695 };
