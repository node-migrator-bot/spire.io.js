 19 var API = require('./spire/api')
 20   , Shred = require('shred')
 21   ;
 22 
 23 /**
 24  * Spire API Client
 25  *
 26  * @class <strong>Spire API Client</strong>
 27  *
 28  * @example
 29  * var spire = new Spire();
 30  *
 31  * @example
 32  * var spire = new Spire({
 33  *   secret: my_account_secret
 34  * });
 35  *
 36  * @constructor
 37  * @param {object} [opts] Options for Spire
 38  * @param {string} [secret] The account API secret.  If you do you not set this, you
 39  * must call one of:
 40  *   * `spire.start(secret, callback)`
 41  *   * `spire.login(email, password, callback)` or
 42  *   * `spire.register(user, callback)
 43  *   before you can start creating channels.
 44  * @param {string} [opts.url] Spire url do use (defaults to 'https://api.spire.io')
 45  * @param {string} opts.version Version of Spire api to use (defaults to '1.0')
 46  * @param {number} opts.timeout Timeout for requests (defaults to 30 seconds)
 47  * @param {boolean} opts.logCurl Log all requests as curl commands (defaults to false)
 48  */
 49 function Spire(opts) {
 50   opts = opts || {};
 51   this.api = new API(this, opts);
 52   this.session = null;
 53   this.shred = new Shred({
 54     logCurl: opts.logCurl
 55   });
 56 }
 57 
 58 module.exports = Spire;
 59 
 60 /**
 61  * Start the Spire session with the given account secret.
 62  *
 63  * @example
 64  * var spire = new Spire();
 65  * spire.start(your_api_secret, function (err, session) {
 66  *   if (!err) {
 67  *     // `session` is a spire session.
 68  *     // Start creating channels and subscripions.
 69  *   }
 70  * });
 71  *
 72  * @param {string} secret The acccount secret
 73  * @param {function(err, session)} cb Callback
 74  */
 75 Spire.prototype.start = function (secret, cb) {
 76   var spire = this;
 77   this.api.createSession(secret, function (err, session) {
 78     if (err) return cb(err);
 79     spire.session = session;
 80     cb(null, session);
 81   });
 82 };

 84 /**

 93  *   }
 94  * });
 95  *

 97  * @param {string} password Password
 98  * @param {function(err, session)} cb Callback
 99  */
100 Spire.prototype.login = function (email, password, cb) {

103     if (err) return cb(err);

109 /**
110  * Retrieve an application with a given application key.
111  *
112  * @example
113  * var spire = new Spire();
114  * spire.getApplication('key-for-your-app', function (err, application) {
115  *   if (!err) {
116  *     // 'application' will now hold your application object.
117  *   }
118  * });
119  *
120  * @param {string} application_key Application key
121  * @param {function(err, application)} cb Callback
122  */
123 Spire.prototype.getApplication = function (application_key, cb) {
124   var spire = this;
125   this.api.getApplication(application_key, function (err, application) {
126     if (err) return cb(err);
127     cb(null, application);
128   });
129 };

131 /**
132  * Registers for a new spire account, and authenticates as the newly created account
133  *
134  * @example
135  * var spire = new Spire();
136  * spire.register({
137  *   email: your_email,
138  *   password: your_password,
139  *   password_confirmation: your_password_confirmation
140  * }, function (err) {
141  *   if (!err) {
142  *     // Your account has been registered.
143  *     // `session` is a spire session.
144  *     // Start creating channels and subscripions.
145  *   }
146  * });
147  *
148  * @param {object} user User info
149  * @param {string} user.email Email
150  * @param {string} user.password Password
151  * @param {string} [user.password_confirmation] Optional password confirmation
152  * @param {function (err, session)} cb Callback
153  */
154 Spire.prototype.register = function (user, cb) {
155   var spire = this;
156   this.api.createAccount(user, function (err, session) {
157     if (err) return cb(err);
158     spire.session = session;
159     cb(null, session);
160   });
161 };

163 /**
164  * Requests a password reset for email.
165  *
166  * @example
167  * var spire = new Spire();
168  * spire.passwordResetRequest(your_email, function (err) {
169  *   if (!err) {
170  *     // A password reset email has been sent.
171  *   }
172  * });
173  *
174  * @param {string} email Email
175  * @param {function (err)} cb Callback
176  */
177 Spire.prototype.passwordResetRequest = function (email, cb) {
178   this.api.passwordResetRequest(email, cb);
179 };
