1 /**
  2  * @fileOverview Channel Resource class definition
  3  */
  4 var Resource = require('./resource')
  5   , Message = require('./message')
  6   , Subscription = require('./subscription')
  7   ;
  8 
  9 /**
 10  * Represents a channel in the spire api.
 11  *
 12  * @class Channel Resource
 13  *
 14  * @constructor
 15  * @extends Resource
 16  * @param {object} spire Spire object
 17  * @param {object} data  Channel data from the spire api
 18  */
 19 function Channel(spire, data) {
 20   this.spire = spire;
 21   this.data = data;
 22   this.resourceName = 'channel';
 23 }
 24 
 25 Channel.prototype = new Resource();
 26 
 27 module.exports = Channel;
 28 
 29 /**
 30  * Returns the channel name.
 31  *
 32  * @returns {string} Channel name
 33  */
 34 Channel.prototype.name = function () {
 35   return this.data.name;
 36 };
 37 
 38 /**
 39  * Publishes a message to the channel.
 40  *
 41  * The messages can be a string, or any json'able object.
 42  *
 43  * @example
 44  * spire.channel('myChannel', function (err, channel) {
 45  *   channel.publish('hello world', function (err, message) {
 46  *     if (!err) {
 47  *       // Message has been published.
 48  *     }
 49  *   });
 50  * });
 51  *
 52  * @param {string|object} message Message to publish
 53  * @param {function (err, message)} cb Callback
 54  */
 55 Channel.prototype.publish = function (message, cb) {
 56   cb = cb || function (err) { if (err) { throw err; } };
 57   var spire = this.spire;
 58   this.request('publish', { content: message }, function (err, messageData) {
 59     if (err) return cb(err);
 60     cb(null, new Message(spire, messageData));
 61   });
 62 };
 63 
 64 /**
 65  * Gets a subscription to a channel, creating it if necessary.
 66  *
 67  * @example
 68  * spire.channel('myChannel', function (err, channel) {
 69  *   channel.subscription('mySubscription', function (err, subscription) {
 70  *     if (!err) {
 71  *       // `subscription` is the new subscription.
 72  *     }
 73  *   });
 74  * });
 75  *
 76  * @param {string} subName Subscription name
 77  * @param {function (err, subscription)} cb Callback
 78  */
 79 Channel.prototype.subscription = function (subName, cb) {
 80   if (!cb) {
 81     cb = subName;
 82     subName = null;
 83   }
 84   var spire = this.spire;
 85   this.request('create_subscription', subName, function (err, subData) {
 86     if (err) return cb(err);
 87     cb(null, new Subscription(spire, subData));
 88   });
 89 };
 90 
 91 /**
 92  * Gets all named subscriptions to a channel, using cached data if available.
 93  *
 94  * @example
 95  * channel.subscriptions(function (err, subscriptions) {
 96 *   if (!err) {
 97 *     // `subscriptions` is a hash of subscriptions
 98 *   }
 99  * });
100  *
101  * @param {function (err, subscriptions)} cb Callback
102  */
103 Channel.prototype.subscriptions = function (cb) {
104   if (this._subscriptions) {
105     return cb(null, this._subscriptions);
106   }
107   this.subscriptions$(cb);
108 };
109 
110 /**
111  * Gets all named subscriptions to a channel, ignoring any cached data.
112  *
113  * @example
114  * channel.subscriptions(function (err, subscriptions) {
115 *   if (!err) {
116 *     // `subscriptions` is a hash of subscriptions
117 *   }
118  * });
119  *
120  * @param {function (err, subscriptions)} cb Callback
121  */
122 Channel.prototype.subscriptions = function (cb) {
123   var channel = this;
124   this.request('subscriptions', function (err, subscriptions) {
125     if (err) return cb(err);
126     channel._subscriptions = subscriptions;
127     cb(null, subscriptions);
128   });
129 };

131 /**
132  * Requests
133  *
134  * These define API calls and have no side effects.  They can be run by calling
135  *     this.request(<request name>);
136  */
137 
138 /**
139  * @name publish
140  * @ignore
141  * Publishes a message to the channel.
142  */
143 Resource.defineRequest(Channel.prototype, 'publish', function (message) {
144   return {
145     method: 'post',
146     url: this.url(),
147     headers: {
148       'Authorization': this.authorization('publish'),
149       'Accept': this.mediaType('message'),
150       'Content-Type': this.mediaType('message')
151     },
152     content: message
153   };
154 });
155 
156 /**
157  * @name subscriptions
158  * @ignore
159  * Gets all subscriptions for a channel.
160  */
161 Resource.defineRequest(Channel.prototype, 'subscriptions', function (message) {
162   var collection = this.data.resources.subscriptions;
163   return {
164     method: 'get',
165     url: collection.url,
166     headers: {
167       'Authorization': this.authorization('get_subscriptions', collection),
168       'Accept': this.mediaType('subscriptions'),
169     }
170   };
171 });
172 
173 /**
174  * @name create_subscription
175  * @ignore
176  * Create a subscription for the channel.
177  */
178 Resource.defineRequest(Channel.prototype, 'create_subscription', function (name) {
179   var collection = this.data.resources.subscriptions;
180   return {
181     method: 'post',
182     url: collection.url,
183     headers: {
184       'Authorization': this.authorization('create', collection),
185       'Accept': this.mediaType('subscription'),
186       'Content-Type': this.mediaType('subscription')
187     },
188     content: {name: name}
189   };
190 });
