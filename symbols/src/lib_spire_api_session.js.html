 15 /**
 16  * Represents a session in the spire api.
 17  *
 18  * <p>Sessions contain other resources, like channels and subscriptions.  These
 19  * can be accessed in the <code>session.resources</code> object.</p>
 20  *
 21  * <p>One important resource inside <code>session</code> is the account resource
 22  * <code>session.account</code>, which is only given to sessions authenticated
 23  * with an email and password.  The account resource is documented in its own
 24  * class.</p>
 25  *
 26  * <p>Session objects maintain lists of channels and subscriptions.  If you call
 27  * the <code>session.channels</code> or <code>session.subscriptions</code>
 28  * methods, you will get back cached data if it exists.  Use the <code>$</code>
 29  * cache-bypass methods: <code>session.channels$</code> and
 30  * <code>session.subscriptions$</code> to get fresh data from the api.</p>
 31  *
 32  * @class Session Resource
 33  *
 34  * @constructor
 35  * @extends Resource
 36  * @param {object} spire Spire object
 37  * @param {object} data Session data from the spire api
 38  */
 39 function Session(spire, data) {
 40   this.spire = spire;
 41   this.data = data;
 42   this.resourceName = 'session';
 43 
 44   this._channels = {};
 45   this._subscriptions = {};
 46   this._applications = {};
 47 	this._storeResources();
 48 
 49 }
 50 
 51 Session.prototype = new Resource();
 52 
 53 module.exports = Session;

 63     if (err) return cb(err);
 64     session.data = data;
 65     session._storeResources();
 66     cb(null, session);
 67   });
 68 };

403 /**
404  * Find or creates a channel
405  *
406  * @example
407  * session.findOrCreateChannel('foo', function (err, channel) {
408  *   if (!err) {
409  *     // `channel` is the channel named "foo".
410  *   }
411  * });
412  *
413  * @param {string} name Channel name to get or create
414  * @param {number} [limit] Number of messages to keep in channel
415  * @param {function(err, channel)} cb Callback
416  */
417 Session.prototype.findOrCreateChannel = function (name, limit, cb) {
418   var session = this;
419   var spire = this.spire;
420 
421   if (!cb) {
422     cb = limit;
423     limit = null;
424   }
425 
426   if (session._channels[name]) {
427     return cb(null, session._channels[name]);
428   }
429 
430   var creationCount = 0;
431 
432   function createChannel() {
433     creationCount++;
434     session.createChannel(name, limit, function (err, channel) {
435       if (!err) return cb(null, channel);
436       if (err.status !== 409) return cb(err);
437       if (creationCount >= spire.CREATION_RETRY_LIMIT) {
438         return cb(new Error("Could not create channel: " + name));
439       }
440       getChannel();
441     });
442   }
443 
444   function getChannel() {
445     spire.session.channelByName(name, function (err, channel) {
446       if (err && err.status !== 404) return cb(err);
447       if (channel) return cb(null, channel);
448       createChannel();
449     });
450   }
451 
452   getChannel();
453 };

455 /**
456  * Creates a subscription to any number of channels.  Returns a Subscription
457  * resource.  Errors if a subscription with the specified name exists.
458  *
459  * @example
460  * session.findOrCreateSubscription({
461  *   name: 'my-sub',
462  *   channelNames: ['foo', 'bar']
463  * },
464  * function (err, subscription) {
465  *   if (!err) {
466  *     // `subscription` is the new subscription
467  *   }
468  * });
469  *
470  * @param {object} options Options
471  * @param {string} options.name Subscription name
472  * @param {array} options.channelNames Channel names to subscribe to
473  * @param {array} options.channelUrls Channel urls to subscribe to
474  * @param {number} options.expiration Subscription expiration (ms)
475  * @param {function (err, subscription)} cb Callback
476  */
477 Session.prototype.createSubscription = function (options, cb) {
478   var session = this;
479 
480   var name = options.name;
481   var channelNames = options.channelNames || [];
482   var channelUrls = options.channelUrls || [];
483   var expiration = options.expiration;
484 
485   function createSubscription() {
486     session.channels(function (channels) {
487       channelUrls.push.apply(channelUrls, _.map(channelNames, function (name) {
488         return session._channels[name].url();
489       }));
490       session.request('create_subscription', name, channelUrls, expiration, function (err, sub) {
491         if (err) return cb(err);
492         var subscription = new Subscription(session.spire, sub);
493         session._memoizeSubscription(subscription);
494         cb(null, subscription);
495       });
496     });
497   }
498 
499   if (channelNames.length) {
500     async.forEach(
501       channelNames,
502       function (channelName, innerCB) {
503         session.findOrCreateChannel(channelName, innerCB);
504       },
505       function (err) {
506         if (err) return cb(err);
507         createSubscription();
508       }
509     );
510   } else {
511     createSubscription();
512   }
513 };

515 /**
516  * Gets a subscription to the given channels.  Creates the channels and the
517  * subscription if necessary.
518  *
519  * @example
520  * session.findOrCreateSubscription('mySubscription', ['foo', 'bar'], function (err, subscription) {
521  *   if (!err) {
522  *     // `subscription` is a subscription named 'mySubscription', listening on channels named 'foo' and 'bar'.
523  *   }
524  * });
525  *
526  * @param {string} Subscription name
527  * @param {array or string} channelOrChannels Either a single channel name, or an array of
528  *   channel names to subscribe to
529  * @param {function (err, subscription)} cb Callback
530  */
531 Session.prototype.findOrCreateSubscription = function (options, cb) {
532   var name = options.name;
533 
534   var session = this;
535   var spire = this.spire;
536 
537   var creationCount = 0;
538 
539   function createSubscription() {
540     creationCount++;
541     session.createSubscription(options, function (err, sub) {
542       if (!err) return cb(null, sub);
543       if (err.status !== 409) return cb(err);
544       if (creationCount >= spire.CREATION_RETRY_LIMIT) {
545         return cb(new Error("Could not create subscription: " + name));
546       }
547       getSubscription();
548     });
549   }
550 
551   function getSubscription() {
552     session.subscriptionByName(options.name, function (err, subscription) {
553       if (err && err.status !== 404) return cb(err);
554       if (subscription) return cb(null, subscription);
555       createSubscription();
556     });
557   }
558 
559   createSubscription();
560 };

562 /**
563  * Creates an new subscription to a channel or channels, and adds a listener.
564  *
565  * @example
566  * session.subscribe('myChannel', options, function (messages) {
567  *   // `messages` is array of messages sent to the channel
568  * }, function (err) {
569  *   // `err` will be non-null if there was a problem creating the subscription.
570  * });
571  *
572  * By default this will get all events from the beginning of time.
573  * If you only want messages created from this point forward, pass { last: 'now' } in the options:
574  *
575  * @example
576  * session.subscribe('myChannel', options, function (messages) {
577  *   // `messages` is array of messages sent to the channel
578  * }, function (err) {
579  *   // `err` will be non-null if there was a problem creating the subscription.
580  * });
581  *
582  * @param {array or string} channelOrChannels Either a single channel name, or an array of
583  *   channel names to subscribe to
584  * @param {object} [options] Options to pass to the listener
585  * @param {function (messages)} listener Listener that will get called with each batch of messages
586  * @param {function (err, subscription)} [cb] Callback
587  */
588 Session.prototype.subscribe = function (channelOrChannels, options, listener, cb) {
589   if (typeof options === 'function') {
590     cb = listener;
591     listener = options;
592     options = {}
593   }
594 
595   if (typeof channelOrChannels === "string") {
596     channels = [channelOrChannels];
597   } else {
598     channels = channelOrChannels;
599   }
600 
601   cb = cb || function () {};
602 
603   this.findOrCreateSubscription({
604     channelNames: channels
605   }, function (err, subscription) {
606     if (err) return cb(err);
607     subscription.addListener('messages', listener);
608     subscription.startListening(options);
609     process.nextTick(function () {
610       cb(null, subscription);
611     });
612   });
613 };

665 /**
666  * Stores the resources.
667  */
668 Session.prototype._storeResources = function () {
669   var session = this;
670 	var resources = {};
671   _.each(this.data.resources, function (resource, name) {
672     // Turn the account object into an instance of Resource.
673     if (name === 'account') {
674       resource = new Account(session.spire, resource);
675       session._account = resource;
676     }
677     resources[name] = resource;
678   });
679 
680   this.resources = resources;
681 };
