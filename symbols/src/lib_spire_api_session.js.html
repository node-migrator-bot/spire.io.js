 14 /**
 15  * Represents a session in the spire api.
 16  *
 17  * <p>Sessions contain other resources, like channels and subscriptions.  These
 18  * can be accessed in the <code>session.resources</code> object.</p>
 19  *
 20  * <p>One important resource inside <code>session</code> is the account resource
 21  * <code>session.account</code>, which is only given to sessions authenticated
 22  * with an email and password.  The account resource is documented in its own
 23  * class.</p>
 24  *
 25  * <p>Session objects maintain lists of channels and subscriptions.  If you call
 26  * the <code>session.channels</code> or <code>session.subscriptions</code>
 27  * methods, you will get back cached data if it exists.  Use the <code>$</code>
 28  * cache-bypass methods: <code>session.channels$</code> and
 29  * <code>session.subscriptions$</code> to get fresh data from the api.</p>
 30  *
 31  * @class Session Resource
 32  *
 33  * @constructor
 34  * @extends Resource
 35  * @param {object} spire Spire object
 36  * @param {object} data Session data from the spire api
 37  */
 38 function Session(spire, data) {
 39   this.spire = spire;
 40   this.data = data;
 41   this.resourceName = 'session';
 42 
 43   this._channels = {};
 44   this._subscriptions = {};
 45   this._applications = {};
 46 	this._storeResources();
 47 
 48 }
 49 
 50 Session.prototype = new Resource();
 51 
 52 module.exports = Session;
 53 
 54 /**
 55  * <p>Gets the Session resource.
 56  *
 57  * @param {function (err, session)} cb Callback
 58  */
 59 Session.prototype.get = function (cb) {
 60   var session = this;
 61   this.request('get', function (err, data) {
 62     if (err) return cb(err);
 63     session.data = data;
 64     session._storeResources();
 65     cb(null, session);
 66   });
 67 };

 69 /**
 70  * Gets the account resource.  Only available to sessions that are authenticated
 71  * with an email and password.
 72  *
 73  * Returns a value from the cache, if one if available.
 74  *
 75  * @example
 76  * spire.session.account(function (err, account) {
 77  *   if (!err) {
 78  *     // `account` is account resource.
 79  *   }
 80  * });
 81  *
 82  * @param {function (err, account)} cb Callback
 83  */
 84 Session.prototype.account = function (cb) {
 85   if (this._account) return cb(null, this._account);
 86   this.account$(cb);
 87 };

 89 /**
 90  * Gets the account resource.  Only available to sessions that are authenticated
 91  * with an email and password.
 92  *
 93  * Always gets a fresh value from the api.
 94  *
 95  * @example
 96  * spire.session.account$(function (err, account) {
 97  *   if (!err) {
 98  *     // `account` is account resource.
 99  *   }
100  * });
101  * @param {function (err, account)} cb Callback
102  */
103 Session.prototype.account$ = function (cb) {
104   var session = this;
105   this.request('account', function (err, account) {
106     if (err) return cb(err);
107     session._account = new Account(session.spire, account);
108     cb(null, session._account);
109   });
110 };

112 /**
113  * Resets the account resource.  Only available to sessions that are authenticated
114  * with an email and password.
115  * *
116  * @example
117  * spire.session.resetAccount(function (err, session) {
118  *   if (!err) {
119  *     // `session` is session with new account resource.
120  *   }
121  * });
122  * @param {function (err, session)} cb Callback
123  */
124 Session.prototype.resetAccount = function (cb) {
125   var session = this;
126   this._account.reset(function (err, sessionData) {
127     if (err) return cb(err);
128 		session.data = sessionData;
129 		session._storeResources();
130     cb(null, session);
131   });
132 };

153 /**
154  * Gets the applications collection.  Returns a hash of Application resources.
155  *
156  * Always gets a fresh value from the api.
157  *
158  * @example
159  * spire.session.applications$(function (err, applications) {
160  *   if (!err) {
161  *     // `applications` is a hash of all the account's applications
162  *   }
163  * });
164  *
165  * @param {function (err, applications)} cb Callback
166  */
167 Session.prototype.applications$ = function (cb) {
168   var session = this;
169   this.request('applications', function (err, applicationsData) {
170     if (err) return cb(err);
171     _.each(applicationsData, function (application, name) {
172       session._memoizeApplication(new Application(session.spire, application));
173     });
174     cb(null, session._applications);
175   });
176 };

359 /**
360  * Creates a channel.  Returns a Channel resource.  Errors if a channel with the
361  * specified name exists.
362  *
363  * @param {string} name Channel name
364  * @param {number} limit Number of messages to keep in channel
365  * @param {function (err, channel)} cb Callback
366  */
367 Session.prototype.createChannel = function (name, limit, cb) {
368   var session = this;
369   if (!cb) {
370     cb = limit;
371     limit = null;
372   }
373 
374   var opts = {
375     name: name
376   }
377 
378   if (limit !== null) {
379     opts.limit = limit
380   }
381 
382   this.request('create_channel', opts, function (err, data) {
383     if (err) return cb(err);
384     var channel = new Channel(session.spire, data);
385     session._memoizeChannel(channel);
386     cb(null, channel);
387   });
388 };

390 /**
391  * Creates a subscription to any number of channels.  Returns a Subscription
392  * resource.  Errors if a subscription with the specified name exists.
393  *
394  * @param {object} options Options
395  * @param {string} options.name Subscription name
396  * @param {array} options.channelNames Channel names to subscribe to
397  * @param {array} options.channelUrls Channel urls to subscribe to
398  * @param {number} options.expiration Subscription expiration (ms)
399  * @param {function (err, subscription)} cb Callback
400  */
401 Session.prototype.createSubscription = function (options, cb) {
402   var name = options.name;
403   var channelNames = options.channelNames || [];
404   var channelUrls = options.channelUrls || [];
405   var expiration = options.expiration;
406 
407   var session = this;
408   this.channels(function (channels) {
409     channelUrls.push.apply(channelUrls, _.map(channelNames, function (name) {
410       return session._channels[name].url();
411     }));
412     session.request('create_subscription', name, channelUrls, expiration, function (err, sub) {
413       if (err) return cb(err);
414       var subscription = new Subscription(session.spire, sub);
415       session._memoizeSubscription(subscription);
416       cb(null, subscription);
417     });
418   });
419 };

448 /**
449  * Stores the resources.
450  */
451 Session.prototype._storeResources = function () {
452   var session = this;
453 	var resources = {};
454   _.each(this.data.resources, function (resource, name) {
455     // Turn the account object into an instance of Resource.
456     if (name === 'account') {
457       resource = new Account(session.spire, resource);
458       session._account = resource;
459     }
460     resources[name] = resource;
461   });
462 
463   this.resources = resources;
464 };
